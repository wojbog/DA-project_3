    data_set = CreateDataLoader(data_test, data_test_target)

    model.eval()
    with torch.no_grad():
        for x, y in data_set:
            output = model(x)
            y_pred_alt = (output > 0.0)
    return y_pred_alt

def predict(alternatives):
    alternatives = torch.tensor(alternatives, dtype=torch.float32)

    model.eval()
    with torch.no_grad():
        output = model(alternatives)
        output = output.reshape(output.shape[0],1)
        probabilities = torch.sigmoid(output).numpy()
        
    return np.hstack([1 - probabilities, probabilities])

from sklearn.inspection import permutation_importance


def predict_fn(alternatives):
    alternatives = torch.tensor(alternatives, dtype=torch.float32)

    model.eval()
    with torch.no_grad():
        output = model(alternatives)
        arr = output.numpy().reshape(-1, 1)
        scaler = MinMaxScaler()
        output = scaler.fit_transform(arr)
        y_pred = (output > 0.5)
    
    return y_pred


class SklearnWrapper:
    def __init__(self, predict_fn):
        self.predict_fn = predict_fn

    def predict(self, X):
        return self.predict_fn(X)

    def fit(self, X, y=None):
        return self


data_test = X_train.reshape(-1, num_criteria)

model_wrapper = SklearnWrapper(predict_fn)
result = permutation_importance(
    model_wrapper,
    data_test,
    predict_fn(data_test),
    scoring='r2',
    n_repeats=10,
    random_state=42,
    n_jobs=-1
)


sorted_result = result.importances_mean.argsort()[::-1]
plt.barh(range(len(sorted_result)), result.importances_mean[sorted_result], align="center")
plt.yticks(range(len(sorted_result)), np.array(feature_names)[sorted_result])
plt.xlabel("Permutation Importance")
plt.title("Permutation Importance of Features")
plt.show()

The graph shows the importance of the features based on the model. The results show similar conclusions with respects to previous analysis.
